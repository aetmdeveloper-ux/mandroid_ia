async function sendMessage() { const text = messageInput.value.trim(); if (!text) return; // Adiciona sua mensagem na tela appendMessage('user', text); messageInput.value = ''; try { const response = await fetch('/api/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ message: text }) // Garanta que a chave é "message" }); const data = await response.json(); if (data.success) { appendMessage('bot', data.message); } else { showToast('⚠ ERRO NA RESPOSTA'); } } catch (err) { console.error(err); showToast('⚠ ERRO DE CONEXÃO'); } }
